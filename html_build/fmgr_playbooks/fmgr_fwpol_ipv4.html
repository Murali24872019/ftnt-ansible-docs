    fmgr_fwpol_ipv4:
      mode: "set"
      adom: "root"
      package_name: "default"
      name: "Test_FSSO_IPv4_Policy"
      comments: "Created by Ansible"

      dstaddr: "srcaddr2"
      srcaddr: "srcaddr1"
      dstintf: "wan1"
      srcintf: "lan"

      schedule: "always"
      fsso: "enable"
      groups: "SSL_OMG_GR"
